    Star one comment.

        star: function(id, star, callback) {
        Comment.findOne({
            _id: id
        }, function(err, comment) {
            if (err) {
                callback(err);
            }
            if (!comment) {
                callback(new Error('Comment not exists!'));
            }
            comment.star = star;
            comment.save(callback);
        });
    },

        saveComment: function(userId, cardId, content, callback) {
